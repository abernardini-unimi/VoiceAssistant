// PADModelConfig mappa la struttura del config.json
type PADModelConfig struct {
	Id2Label map[string]string `json:"id2label"`
}

// PADManager gestisce l'analisi emotiva in background e mantiene lo stato attuale
type PADManager struct {
	audioMu          sync.Mutex
	audioBuf         []float32
	samplesSinceLast int 
	isSpeaking       bool // Flag per decidere se calcolare o no

	stateMu         sync.Mutex
	currentPADState string 

	onnxPath   string
	configPath string
	config     PADModelConfig

	session *ort.DynamicAdvancedSession
	options *ort.SessionOptions

	initOnce sync.Once
	isReady  bool
}

func NewPADManager(onnxPath, configPath string) *PADManager {
	fileConfig, err := os.ReadFile(configPath)
	if err != nil {
		log.Printf("[PAD] Errore lettura config: %v", err)
		return nil
	}
	var config PADModelConfig
	if err := json.Unmarshal(fileConfig, &config); err != nil {
		log.Printf("[PAD] Errore parsing config: %v", err)
		return nil
	}
	return &PADManager{
		onnxPath:   onnxPath,
		configPath: configPath,
		config:     config,
		isSpeaking: false,
	}
}

func (m *PADManager) InitSession() {
	m.initOnce.Do(func() {
		opts, err := ort.NewSessionOptions()
		if err != nil {
			log.Printf("[PAD] Errore creazione opzioni ONNX: %v", err)
			return
		}

		session, err := ort.NewDynamicAdvancedSession(
			m.onnxPath,
			[]string{"input_values"},
			[]string{"logits"},
			opts,
		)
		if err != nil {
			log.Printf("[PAD] Errore avvio ONNX: %v", err)
			opts.Destroy()
			return
		}

		m.session = session
		m.options = opts
		m.isReady = true
		log.Println("[PAD] 🚀 Sessione ONNX in memoria Inizializzata. Motore indipendente avviato!")
	})
}

func (m *PADManager) SetSpeaking(speaking bool) {
	m.audioMu.Lock()
	defer m.audioMu.Unlock()
	m.isSpeaking = speaking
}

// ProcessAudioChunk tiene sempre gli ultimi 3 secondi, ma calcola solo se isSpeaking = true
func (m *PADManager) ProcessAudioChunk(pcm []float32) {
	if !m.isReady {
		return
	}

	m.audioMu.Lock()
	m.audioBuf = append(m.audioBuf, pcm...)
	m.samplesSinceLast += len(pcm)
	
	// Finestra Scorrevole SEMPRE ATTIVA: mantieni al massimo gli ultimi 3 secondi
	if len(m.audioBuf) > 48000 {
		m.audioBuf = m.audioBuf[len(m.audioBuf)-48000:]
	}

	shouldRun := false
	var chunk []float32

	// TRIGGER: Se stiamo parlando, ogni secondo spariamo l'inferenza in background
	if m.isSpeaking && m.samplesSinceLast >= 16000 && len(m.audioBuf) >= 16000 {
		shouldRun = true
		m.samplesSinceLast = 0
		chunk = make([]float32, len(m.audioBuf))
		copy(chunk, m.audioBuf)
	}
	m.audioMu.Unlock()

	if shouldRun {
		go m.runInference(chunk)
	}
}

// FlushAudio viene chiamato allo stop del VAD. Calcola l'ultimo residuo.
func (m *PADManager) FlushAudio() {
	if !m.isReady {
		return
	}

	m.audioMu.Lock()
	chunk := make([]float32, len(m.audioBuf))
	copy(chunk, m.audioBuf)
	m.audioBuf = nil // Svuota il buffer per la prossima frase
	m.samplesSinceLast = 0
	m.audioMu.Unlock()

	if len(chunk) >= 8000 {
		go m.runInference(chunk)
	}
}

func (m *PADManager) runInference(audioFloat []float32) {
	var sum float64
	for _, val := range audioFloat {
		sum += float64(val)
	}
	mean := sum / float64(len(audioFloat))

	var variance float64
	for _, val := range audioFloat {
		diff := float64(val) - mean
		variance += diff * diff
	}
	variance /= float64(len(audioFloat))
	stdDev := float32(math.Sqrt(variance + 1e-7))

	for i := range audioFloat {
		audioFloat[i] = (audioFloat[i] - float32(mean)) / stdDev
	}

	inputShape := ort.NewShape(1, int64(len(audioFloat)))
	inputTensor, err := ort.NewTensor(inputShape, audioFloat)
	if err != nil { return }
	defer inputTensor.Destroy()

	outputShape := ort.NewShape(1, 3)
	outputTensor, err := ort.NewEmptyTensor[float32](outputShape)
	if err != nil { return }
	defer outputTensor.Destroy()

	err = m.session.Run(
		[]ort.ArbitraryTensor{inputTensor},
		[]ort.ArbitraryTensor{outputTensor},
	)
	if err != nil {
		log.Printf("[PAD] Errore esecuzione modello: %v", err)
		return
	}

	valori := outputTensor.GetData()
	risultatoFinale := make(map[string]float64)
	for i, val := range valori {
		chiaveStr := strconv.Itoa(i)
		label, ok := m.config.Id2Label[chiaveStr]
		if !ok {
			label = fmt.Sprintf("classe_%d", i)
		}
		risultatoFinale[label] = math.Round(float64(val)*10000) / 10000
	}

	jsonOutput, _ := json.MarshalIndent(risultatoFinale, "", "    ")
	jsonLog, _ := json.Marshal(risultatoFinale)

	m.stateMu.Lock()
	m.currentPADState = string(jsonOutput)
	m.stateMu.Unlock()
	
	log.Printf("🎛️ [PAD] Nuovo stato emotivo: %s", string(jsonLog))
}

func (m *PADManager) GetAndClearCurrentState() string {
	m.stateMu.Lock()
	defer m.stateMu.Unlock()
	
	state := m.currentPADState
	m.currentPADState = "" 
	return state
}

func bytesToFloat32(data []byte) []float32 {
	numSamples := len(data) / 2
	floats := make([]float32, numSamples)
	for i := 0; i < numSamples; i++ {
		s := int16(data[i*2]) | int16(data[i*2+1])<<8
		floats[i] = float32(s) / 32768.0
	}
	return floats
}

type PADCollectorElement struct {
	*pipeline.BaseElement
	manager *PADManager
}

func NewPADCollectorElement(m *PADManager) *PADCollectorElement {
	return &PADCollectorElement{
		BaseElement: pipeline.NewBaseElement("pad-collector", 100),
		manager:     m,
	}
}

func (e *PADCollectorElement) Start(ctx context.Context) error {
	startChan := make(chan pipeline.Event, 10)
	stopChan := make(chan pipeline.Event, 10)

	// Controllo di sicurezza: verifichiamo che il Bus esista prima di iscriverci
	bus := e.BaseElement.Bus()
	if bus != nil {
		log.Println("[PAD] ✅ Bus connesso, iscrizione eventi VAD...")
		bus.Subscribe(pipeline.EventVADSpeechStart, startChan)
		bus.Subscribe(pipeline.EventVADSpeechEnd, stopChan)
		//bus.Subscribe(pipeline.EventVADSpeechStop, stopChan) // Fallback di sicurezza
	} else {
		log.Println("[PAD] ❌ ERRORE: Bus non connesso! Il rubinetto non si aprirà.")
	}

	go func() {
		defer close(e.OutChan)
		for {
			select {
			case <-ctx.Done():
				return
			case <-startChan:
				log.Println("[PAD] 🟢 Rubinetto Aperto (Inizio parlato)")
				e.manager.SetSpeaking(true)
			case <-stopChan:
				log.Println("[PAD] 🔴 Rubinetto Chiuso (Fine parlato, calcolo finale...)")
				e.manager.SetSpeaking(false)
				e.manager.FlushAudio()
			case msg, ok := <-e.InChan:
				if !ok { return }
				
				// L'audio passa liberamente senza mai bloccarsi
				if msg.Type == pipeline.MsgTypeAudio && msg.AudioData != nil {
					floats := bytesToFloat32(msg.AudioData.Data)
					e.manager.ProcessAudioChunk(floats)
				}
				
				select {
				case e.OutChan <- msg:
				case <-ctx.Done(): return
				}
			}
		}
	}()
	return nil
}

type PADEnricherElement struct {
	*pipeline.BaseElement
	manager *PADManager
}

func NewPADEnricherElement(m *PADManager) *PADEnricherElement {
	return &PADEnricherElement{
		BaseElement: pipeline.NewBaseElement("pad-enricher", 100),
		manager:     m,
	}
}

func (e *PADEnricherElement) Start(ctx context.Context) error {
	go func() {
		defer close(e.OutChan)
		for {
			select {
			case <-ctx.Done():
				return
			case msg, ok := <-e.InChan:
				if !ok { return }

				if msg.TextData != nil && len(msg.TextData.Data) > 0 {
					padState := e.manager.GetAndClearCurrentState()
					
					// Se Whisper batte il record del mondo (es. 380ms), diamo a ONNX 
					// giusto 250 millisecondi di respiro per fornirci il FlushAudio finale.
					if padState == "" {
						time.Sleep(250 * time.Millisecond)
						padState = e.manager.GetAndClearCurrentState()
					}

					if padState != "" {
						originalText := string(msg.TextData.Data)
						nuovoTesto := fmt.Sprintf("%s\n\n[Analisi Acustica (Arousal, Dominance, Valence)]:\n%s\n(Modula il tono della risposta basandoti su questi parametri)", originalText, padState)
						msg.TextData.Data = []byte(nuovoTesto)
						log.Printf("[PAD] Inserito PAD nel prompt dell'LLM (Latenza Zero)!")
					} else {
						log.Printf("[PAD] Nessun dato PAD disponibile, procedo con la sola trascrizione.")
					}
				}

				select {
				case e.OutChan <- msg:
				case <-ctx.Done(): return
				}
			}
		}
	}()
	return nil
}
